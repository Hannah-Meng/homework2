library(dplyr)
library(ggplot2)
flights2 <- flights %>%
  mutate(
    dep_hour = floor(dep_time/100),
    arr_hour = floor(arr_time/100)
  )
flights2 <- flights2 %>%
  mutate(
    dep_period = case_when(
      dep_hour >= 0 & dep_hour < 6 ~"early morning",
      dep_hour >= 6 & dep_hour < 12 ~"morning", 
      dep_hour >= 12 & dep_hour < 18 ~"afternoon",
      dep_hour >= 18 & dep_hour < 23 ~"evening",
      TRUE ~ as.character(NA)
    ),
    arr_period = case_when(
      arr_hour >= 0 & arr_hour < 6 ~"early morning",
      arr_hour >= 6 & arr_hour < 12 ~"morning",
      arr_hour >= 12 & arr_hour < 18 ~"afternoon",
      arr_hour >= 18 & arr_hour < 23 ~"evening",
      TRUE ~ as.character(NA)
    )
  )

ggplot(flights2, aes(x = dep_period))+
  geom_bar(fill = "blue")+
  labs(title = "Departure Time Periods",
       x = "Departure Period", y = "Number of Flights")+
  theme_minimal()
ggplot(flights2, aes(x = arr_period))+
  geom_bar(fill = "orange")+
  labs(title = "Arrive Time Periods",
       x = "Arrival Period", y = "Number of FLights")+
  theme_minimal()
flights2 <- flights2 %>%
  mutate(redeye = dep_period %in% c("afternoon", "evening") & 
           arr_period %in% c("afternoon", "evening"))

redeye_pct <- mean(flights2$redeye, na.rm =TRUE)*100
